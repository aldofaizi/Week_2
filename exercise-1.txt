2. Tahun Kabisat
    STORE "tahun"

    IF "tahun" divide 4 == 0 AND
       "tahun" divide 100 == 0 AND
       "tahun" divide 400 == 0
        DISPLAY "Ini tahun kabisat"
    ELSE
        DISPLAY "Bukan tahun kabisat"
