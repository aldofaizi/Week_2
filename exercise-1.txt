2. Tahun Kabisat
    STORE "tahun"

    IF "tahun" mod 4 == 0
        THEN
            IF "tahun" mod 100 == 0
                THEN
                    IF "tahun" mod 400 == 0
                    THEN
                    DISPLAY "Ini tahun kabisat"
    ELSE
        DISPLAY "Bukan tahun kabisat"
